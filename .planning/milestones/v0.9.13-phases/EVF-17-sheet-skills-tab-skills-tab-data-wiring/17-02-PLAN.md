     * spectrum (○/◉/★) per UI-SPEC §3, with half-prof rounded up to ◉ at
     * render time (renderer's job, not reader's).
     *
     * `ability` falls back to SKILL_DEFAULT_ABILITY[key] when dnd5e leaves
     * the field absent. `passive` reads dnd5e's prep-time computed value
     * verbatim — NOT recomputed from `10 + total` (magic-item bonuses,
     * Observant feat, etc. may diverge).
     *
     * @internal
     */
    function readSkill(raw: Dnd5eSkillRaw | undefined, key: SkillKey): Skill {
      const total = raw?.total ?? 0;
      const ability = (raw?.ability ?? SKILL_DEFAULT_ABILITY[key]) as AbilityKey;
      // Validate ability is in the 6-key set; fallback to canonical default if drift.
      const safeAbility: AbilityKey =
        ability === 'str' || ability === 'dex' || ability === 'con' ||
        ability === 'int' || ability === 'wis' || ability === 'cha'
          ? ability
          : SKILL_DEFAULT_ABILITY[key];
      const proficient = (raw?.proficient ?? 0) as 0 | 0.5 | 1 | 2;
      // Clamp proficient to valid enum (defensive against malformed input)
      const safeProficient: 0 | 0.5 | 1 | 2 =
        proficient === 0 || proficient === 0.5 || proficient === 1 || proficient === 2
          ? proficient
          : 0;
      const passive = raw?.passive ?? 10;
      // Clamp passive to non-negative (schema requires this — defensive against negative drift)
      const safePassive = passive < 0 ? 0 : passive;
      return { total, ability: safeAbility, proficient: safeProficient, passive: safePassive };
    }
