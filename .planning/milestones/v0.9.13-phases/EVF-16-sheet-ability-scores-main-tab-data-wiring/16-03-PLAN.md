    /**
     * Right-align an ability value in a 2-cell field.
     *
     * 8 → ' 8', 16 → '16', 21 → '21'. Asserts 0 ≤ n ≤ 30 (matches AbilityScoreSchema bound).
     * value=30 would produce '30' (still 2-cell); only an out-of-range value > 99 would

     *
     * @see UI-SPEC §3 (format helpers); CONTEXT §Area 3 (width budget)
     */
    export function formatAbilityValue(n: number): string {
      // Defensive guard — schema clamps 0..30 but the helper stays robust to bad data
      if (n < 0 || n > 99) return '??';
      return n < 10 ? ` ${n}` : String(n);
    }

    /**
     * Always-signed 2-cell modifier string.
     *
     * +3 → '+3', -1 → '-1', 0 → '+0'. For standard D&D 5e value range 0..30,
     * mod is in range -5..+10; the +10 case requires 3 cells (UI-SPEC §3 documents
     * the overflow). Range -9..+9 fits the 2-cell budget guaranteed.
     *
     * @see UI-SPEC §3 (format helpers)
     */
    export function formatAbilityMod(n: number): string {

      return `${n}`; // negative sign already part of the number, e.g. '-1'
