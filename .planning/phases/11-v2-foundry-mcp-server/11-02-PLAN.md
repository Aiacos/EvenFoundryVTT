    **register-tools.ts implementation:**

    ```ts
    import { CastSpellInputSchema, WeaponAttackInputSchema, UseItemInputSchema, MoveTokenInputSchema, PlaceTemplateInputSchema, DropConcentrationInputSchema } from '@evf/shared-protocol';
    import type { McpServer } from '@modelcontextprotocol/sdk/server/mcp.js';
    import type { Logger } from 'pino';
    import type { BridgeClient } from './bridge-client.js';
    import { TOOL_DESCRIPTIONS } from './tool-descriptions.js';

    export const EVF_MCP_TOOL_IDS = ['cast-spell','weapon-attack','use-item','move-token','place-template','drop-concentration'] as const;
    export type EvfMcpToolId = typeof EVF_MCP_TOOL_IDS[number];

    // kebab → snake for bridge tool name lookup
    const TO_SNAKE: Record<EvfMcpToolId, string> = {
      'cast-spell': 'cast_spell',
      'weapon-attack': 'weapon_attack',
      'use-item': 'use_item',
      'move-token': 'move_token',
      'place-template': 'place_template',
      'drop-concentration': 'drop_concentration',
    };

    export function registerEvfTools(server: McpServer, bridgeClient: BridgeClient, logger: Logger): void {
      const TOOL_SHAPES = {
        'cast-spell':         CastSpellInputSchema.shape,
        'weapon-attack':      WeaponAttackInputSchema.shape,
        'use-item':           UseItemInputSchema.shape,
        'move-token':         MoveTokenInputSchema.shape,
        'place-template':     PlaceTemplateInputSchema.shape,
        'drop-concentration': DropConcentrationInputSchema.shape,
      } as const;

      for (const toolId of EVF_MCP_TOOL_IDS) {
        const { title, description } = TOOL_DESCRIPTIONS[toolId];
        server.registerTool(
          toolId,
          { title, description, inputSchema: TOOL_SHAPES[toolId] },
          async (args) => {
            const snakeName = TO_SNAKE[toolId];
            const result = await bridgeClient.invokeTool(snakeName, args);
            // Bridge response is the raw ToolResult from dispatchTool.
            if (result.success) {
              return { content: [{ type: 'text', text: JSON.stringify(result.data ?? {}) }] };
            }
            logger.warn({ toolId, error: result.error }, 'tool.invoke returned failure');
            return { content: [{ type: 'text', text: result.error ?? 'unknown_error' }], isError: true };
          },
        );
      }
      logger.info({ count: EVF_MCP_TOOL_IDS.length }, 'MCP tools registered');
    }
    ```
